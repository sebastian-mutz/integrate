Measuring the Amount of Water¶

Clausius-Clapeyron Ratio¶

\[\begin{equation} \frac {dP}{dT} = \frac {L}{T · \Delta v} = \frac {\Delta s}{\Delta v} \end{equation}\]

Clausius-Clapeyron Equation¶

\[\Delta v \approx v_g\]
\[\begin{equation} P · V = n · \bar{R} · T \end{equation}\]
\[\begin{equation} P · v = \bar{R} · T \end{equation}\]
\[\begin{equation} v_g = \frac {R · T} {P} \end{equation}\]
\[\begin{equation} \frac {dP}{dT} = \frac {L}{T · \Delta v} \end{equation}\]
\[\begin{equation} \frac {dP}{dT} = \frac {L}{T^2 · R} \end{equation}\]

Saturation Vapour Pressure¶

\[\begin{equation} P_s = 610.78 · e^{ \frac {17.2684 · t} {t + 238.3}} \end{equation}\]
\[\begin{equation} P_s = c · e^{ \frac {a · t} {t + b}} \end{equation}\]
\[a = 17.625\]
\[b = 233.04\]
\[c = 610.94\]
\[\begin{equation} P_s = \frac {exp(43.494-\frac{6545.8}{t+237.1}) } {(t+105)^{1.57}} \end{equation}\]

Dew Point Temperature¶

\[\begin{equation} RH = \frac {P_v}{P_s(t)} · 100 \end{equation}\]
\[\begin{equation} P_s = c · e^{ \frac {a · t} {t + b}} \end{equation}\]
\[\begin{equation} t_d = \frac {b · [ln( \frac {RH}{100} ) + \frac {a · t}{b + t} ]} {a - ln ( \frac {RH}{100} ) - \frac {a · t}{b + t}} \end{equation}\]
\[\begin{equation} t_d = t - ( \frac {100 - RH} {5} ) \end{equation}\]
